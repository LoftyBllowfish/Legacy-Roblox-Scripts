function addSound(song)
	for _,v in pairs(song) do
		local s = Instance.new("Sound")
		s.Pitch = 1
		s.Volume = 0.5
		s.SoundId = ""
		s.Parent = script
		v[3] = s
	end
end

addSound(Songs.Song1)
addSound(Songs.Song2)

function getHighest(Song)
	local high = 0
	for _,v in pairs(Song) do
		if #v[2] > high then
			high = #v[2]
		end
	end
	return high
end

function Pla(ID, PITCH, SOUND)
	local Sound = SOUND
	Sound.SoundId = ID
	Sound.Pitch = PITCH
	Sound:play()
end

function PlaySong(Song)
	for i = 1, getHighest(Song) do
		for _,v in pairs(Song) do
			local t = v[2][i]
			if t then
				Pla(
					(t[1]),
					(t[2]), 
					(v[3])
				)
			end
		end
		wait(Tempo)
	end
end

while true do
PlaySong(Songs.Song2)
