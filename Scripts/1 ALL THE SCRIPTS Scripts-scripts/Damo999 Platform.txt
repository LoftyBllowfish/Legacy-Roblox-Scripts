workspace:findFirstChild("nilDL"):Remove() 
end 
local mdl = Instance.new("Model",workspace)
mdl.Name = "nilDL"
p = Instance.new("Part",mdl)
p.Parent = game.workspace
p.Name = "Dark"
p.BrickColor = BrickColor.new("Black")
p.TopSurface = "Smooth"
p.Position = Vector3.new(-14, 1, -89.5)
p.Size = Vector3.new(8, 1.2, 7)
p.Anchored= true
p.CanCollide = true
p.Locked = true
local part = game.Workspace.Dark
function onTouched(hit)
game.Workspace.Dark.BrickColor = BrickColor.new("White")
wait(0.5)
game.Workspace.Dark.BrickColor = BrickColor.new("Black")
wait(0.5)
game.Workspace.Dark.BrickColor = BrickColor.new("Black")
end
part.Touched:connect(onTouched)
