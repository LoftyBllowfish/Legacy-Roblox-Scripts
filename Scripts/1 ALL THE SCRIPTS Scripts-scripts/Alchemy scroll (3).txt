function getTime()
	local tick = math.fmod(tick(),86400)

	local hour = math.floor(tick/3600) + 5 + time_zone
	local minute = math.floor(tick/60 - 60*hour)

	local m = ""
	if (not hour24) then
		if (hour > 12) then
			hour = hour - 12
			m = " PM"
		else
			hour = (hour == 0) and 12 or hour
			m = " AM"
		end
	end

	return string.format("%2.0f:%02.0f",hour,minute)..m
end
