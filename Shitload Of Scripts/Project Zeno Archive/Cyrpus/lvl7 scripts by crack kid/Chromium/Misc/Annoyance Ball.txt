function rollto()
   choice = math.random(#game.Players:GetChildren())
   player = game.Players:GetChildren()[choice]
   if player.Character ~= nil then
       torso = player.Character:FindFirstChild("Torso")
       if torso ~= nil then
           offset = ppart.Position - torso.Position
           ppart.Velocity = -offset/2
       end
   end
end
