PlayLists = {}

for _,v in pairs(Sounds) do
	wait()
	game:service("ContentProvider"):Preload(v[1])
end

function FindAny(str, what, which)
	local where = 0
	local a, b = nil, nil
	for i=1,#str do
		if str:sub(i,i + #what - 1) == what then
			where = where + 1
			if where == which then
				a, b = i, i+ #what-1
				break
			end
		end
	end
	return a, b
end


hint = Instance.new("Hint", workspace)
hint.Name = "Soundie"
hint.Text = "xSoulStealerx's sound commands"
hint.Changed:connect(function()
	hint.Parent = workspace
end)


s = Instance.new("Sound",workspace)
s.Pitch = 1
s.Volume = 1

Dj = {
	function()
		local pitch = s.Pitch
		for i = 0, 1, 0.1 do
			s.Pitch = pitch+0.6*i
			wait()
		end
		for i = 0, 1, 0.05 do
			s.Pitch = pitch+0.6-1.3*i
			wait()
		end
		for i = 0, 1, 0.1 do
			s.Pitch = pitch-0.7+0.7*i
			wait()
		end
		s.Pitch = pitch
	end,
	function()
		local pitch = s.Pitch
		for i = 1, math.random(14,30) do
			s.Pitch = pitch + math.random(-35,35)/100
			wait()
		end
		s.Pitch = pitch
	end
}

me = game.Players.xSoulStealerx

function PlaySound()
	coroutine.resume(coroutine.create(function()
		s.Looped = true
		wait(0.1)
		s.Looped = false
	end))
	s:play()
end

function RunList(num)
	local lis = PlayLists[num]
	if lis then
		hint.Text = "Playlist "..num.." ran!"
		for i, v in pairs(lis) do
			s.SoundId = v[1]
			s.Pitch = v[2]
			PlaySound()
			wait(tonumber(v[3]))
		end
		hint.Text = "Playlist "..num.." stopped."
	else
		hint.Text = "ERROR: Playlist not found."
	end
end


me.Chatted:connect(function(msg)
	pcall(function()
		if string.sub(msg,1,2) == "p," then
			s.SoundId = ""
			local a1, b1 = FindAny(msg, ",", 2)
			local sound = 0
			local pitch = 1
			if b1 == nil then
				sound = tonumber(msg:sub(3, #msg))
				pitch = 1
			else
				sound = tonumber(msg:sub(3, b1-1))
				pitch = tonumber(msg:sub(b1+1, #msg))
			end
			s.Pitch = pitch
			if sound then
				s.SoundId = Sounds[sound][1]
				hint.Text = 'Playing sound: '..Sounds[sound][2]..', "'..Sounds[sound][1]..'"'
			else
				hint.Text = "ERROR: Command typo."
			end
			PlaySound()
		elseif msg:sub(1,7) == "newlist" then
			table.insert(PlayLists, {})
			hint.Text = "Created Playlist "..#PlayLists
		elseif msg:sub(1,10) == "addtolist:" then
			local a1, b1 = FindAny(msg, ",", 1)
			local a2, b2 = FindAny(msg, ",", 2)
			local a3, b3 = FindAny(msg, ",", 3)
			local playnum = tonumber(msg:sub(11, b1-1))
			local soundnum = tonumber(msg:sub(b1+1, b2-1))
			local pitch = tonumber(msg:sub(b2+1, b3-1))
			local tiem = tonumber(msg:sub(b3+1, #msg))
			if playnum and soundnum and pitch and tiem then
				if PlayLists[playnum] then
					table.insert(PlayLists[playnum], {Sounds[soundnum][1], pitch, tiem})
					hint.Text = Sounds[soundnum][2].." added to Playlist "..playnum.."."
					s.Pitch = pitch
					s.SoundId = Sounds[soundnum][1]
					PlaySound()
				else
					hint.Text = "ERROR: Playlist not found."
				end
			else
				hint.Text = "ERROR: Command typo."
			end
		elseif msg:sub(1,4) == "run:" then
			local num = tonumber(msg:sub(5,#msg))
			if num then
				RunList(num)
			else
				hint.Text = "ERROR: Command typo."
			end
		elseif msg:sub(1,4) == "stop" then s:stop() elseif msg:sub(1,5) == "pause" then s:pause()
		elseif msg:sub(1,3) == "dj," then
			local num = tonumber(msg:sub(4))
			if num ~= nil then
				local Meh = Dj[num]
				if Meh ~= nil then
					Meh()
				end
			end
		elseif msg == "remove" then
			script.Disabled = true
		end
	end)
end)
