function FindHumanoid(source)
local object = source
repeat
if object.Parent ~= workspace then
object = object.Parent
else
return true
end
until true
 
if object:findFirstChild("Humanoid") ~= nil then
Damage(object.Humanoid)
end
 
end
 
function Slash(target)

FindHumanoid(target)
end
end
