function ParseMessage(Message)
Message = Message:gsub("lego%s", "")
Message = Message:gsub("runescape%s", "")
Message = Message:gsub("minecraft%s", "")
local Command
local Args = {}
for Word in Message:gmatch("%w+") do
if not Command then
Command = Word
else
table.insert(Args, Word)
end
end
return Command, Args
end
